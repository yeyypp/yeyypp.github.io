#Sorting
- MergeSort  
```java
public class Merge {
    private Merge() {}

    public static void sort(int[] nums) {
        int[] aux = new int[nums.length];
        sort(nums, aux, 0, nums.length - 1);
    }

    private static void sort(int[] nums, int[] aux, int low, int high) {
        if (high <= low) return;
        int mid = low + (high - low) / 2;
        sort(nums, aux, low, mid);
        sort(nums, aux, mid + 1, high);
        merge(nums, aux, low, mid, high);
    }

    private static void merge(int[] nums, int[] aux, int low, int mid, int high) {
        for (int k = low; k <= high; k++) {
            aux[k] = nums[k];
        }

        int i = low, j = mid + 1;
        for (int k = low; k <= high; k++) {
            if (i > mid) nums[k] = aux[j++];
            else if (j > high) nums[k] = aux[i++];
            else if (aux[i] < aux[j]) nums[k] = aux[i++];
            else nums[k] = aux[j++];
        }
    }
}
```

- Quicksort
```java
public class Quick {
    private void Quick() {}

    public static void sort(int[] nums) {
        sort(nums, 0, nums.length - 1);
    }

    private static void sort(int[] nums, int low, int high) {
        if (high <= low) return;
        int j = partition(nums, low, high);
        sort(nums, low, j - 1);
        sort(nums, j + 1, high);
    }

    private static int partition(int[] nums, int low, int high) {
        int i = low, j = high + 1;
        int sentinel = nums[low];
        while (true) {
            while (nums[++i] < sentinel) {
                if (i == high) break;
            }

            while (nums[--j] > sentinel) {
                if (j == low) break;
            }

            if (i >= j) break;

            exch(nums, i, j);
        }

        exch(nums, low, j);
        return j;
    }

    private static void exch(int[] nums, int i, int j) {
        int tem = nums[i];
        nums[i] = nums[j];
        nums[j] = tem;
    }
}
```
